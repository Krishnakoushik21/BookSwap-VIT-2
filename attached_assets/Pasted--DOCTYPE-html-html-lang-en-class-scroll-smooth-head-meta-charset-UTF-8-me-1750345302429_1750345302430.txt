        function initializeDarkMode() {
            const themeToggle = document.getElementById('theme-toggle');
            const mobileThemeToggle = document.getElementById('mobile-theme-toggle');
            
            // Check for saved theme preference or default to light mode
            const currentTheme = localStorage.getItem('theme') || 'light';
            
            if (currentTheme === 'dark') {
                document.documentElement.classList.add('dark');
            } else {
                document.documentElement.classList.remove('dark');
            }
            
            function toggleTheme() {
                if (document.documentElement.classList.contains('dark')) {
                    document.documentElement.classList.remove('dark');
                    localStorage.setItem('theme', 'light');
                } else {
                    document.documentElement.classList.add('dark');
                    localStorage.setItem('theme', 'dark');
                }
            }
            
            themeToggle.addEventListener('click', toggleTheme);
            mobileThemeToggle.addEventListener('click', toggleTheme);
        }

                if (entry.isIntersecting) {
                    const delay = parseInt(entry.target.dataset.delay) || 0;
                    const step = parseInt(entry.target.dataset.step);
                    
                    setTimeout(() => {
                        if (entry.target.classList.contains('animate-on-scroll')) {
                            entry.target.classList.remove('opacity-0');
                            entry.target.classList.add('animate-fade-in-up');
                        }
                        
                        if (step) {
                            entry.target.classList.remove('opacity-0');
                            if (entry.target.classList.contains('step-circle')) {
                                entry.target.classList.add('animate-scale-in');
                            } else {
                                entry.target.classList.add('animate-slide-in-left');
                            }
                            
                            // Animate timeline progress
                            const progress = document.getElementById('timeline-progress');
                            if (progress) {
                                setTimeout(() => {
                                    progress.style.height = `${(step / 4) * 100}%`;
                                }, delay + 300);
                            }
                        }
                    }, delay);
                    
                    scrollObserver.unobserve(entry.target);
                }
            });
        }, { 
            threshold: 0.1,
            rootMargin: '0px 0px -50px 0px'
        });

        // Enhanced testimonial carousel
        function initializeTestimonialCarousel() {
            const track = document.getElementById('testimonial-track');
            const prevBtn = document.getElementById('prev-testimonial');
            const nextBtn = document.getElementById('next-testimonial');
            const indicators = document.querySelectorAll('.testimonial-indicator');

            function updateCarousel() {
                const translateX = -currentTestimonial * 100;
                track.style.transform = `translateX(${translateX}%)`;
                
                indicators.forEach((indicator, index) => {
                    if (index === currentTestimonial) {
                        indicator.classList.remove('bg-gray-300', 'dark:bg-gray-600');
                        indicator.classList.add('bg-royal');
                    } else {
                        indicator.classList.remove('bg-royal');
                        indicator.classList.add('bg-gray-300', 'dark:bg-gray-600');
                    }
                });
            }

            function nextTestimonial() {
                currentTestimonial = (currentTestimonial + 1) % testimonialCount;
                updateCarousel();
            }

            function prevTestimonial() {
                currentTestimonial = (currentTestimonial - 1 + testimonialCount) % testimonialCount;
                updateCarousel();
            }

            function startAutoplay() {
                testimonialAutoplay = setInterval(nextTestimonial, 6000);
            }

            function stopAutoplay() {
                if (testimonialAutoplay) {
                    clearInterval(testimonialAutoplay);
                }
            }

            // Event listeners with enhanced animations
            nextBtn.addEventListener('click', () => {
                stopAutoplay();
                nextTestimonial();
                nextBtn.style.transform = 'scale(0.9)';
                setTimeout(() => nextBtn.style.transform = 'scale(1)', 150);
                startAutoplay();
            });

            prevBtn.addEventListener('click', () => {
                stopAutoplay();
                prevTestimonial();
                prevBtn.style.transform = 'scale(0.9)';
                setTimeout(() => prevBtn.style.transform = 'scale(1)', 150);
                startAutoplay();
            });

            indicators.forEach((indicator, index) => {
                indicator.addEventListener('click', () => {
                    stopAutoplay();
                    currentTestimonial = index;
                    updateCarousel();
                    startAutoplay();
                });
            });

            const carousel = document.getElementById('testimonial-carousel');
            carousel.addEventListener('mouseenter', stopAutoplay);
            carousel.addEventListener('mouseleave', startAutoplay);

            startAutoplay();
        }

        // Enhanced sign-in modal
        function initializeSignInModal() {
            const modal = document.getElementById('signin-modal');
            const modalContent = document.getElementById('modal-content');
            const closeBtn = document.getElementById('close-modal');
            const signinBtns = [
                document.getElementById('nav-signin-btn'),
                document.getElementById('mobile-signin-btn'),
                document.getElementById('join-btn'),
                document.getElementById('footer-cta-btn')
            ];

            function openModal() {
                modal.classList.remove('hidden');
                modal.classList.add('flex');
                setTimeout(() => {
                    modalContent.classList.remove('scale-95', 'opacity-0');
                    modalContent.classList.add('scale-100', 'opacity-100');
                }, 10);
            }

            function closeModal() {
                modalContent.classList.remove('scale-100', 'opacity-100');
                modalContent.classList.add('scale-95', 'opacity-0');
                setTimeout(() => {
                    modal.classList.add('hidden');
                    modal.classList.remove('flex');
                }, 300);
            }

            signinBtns.forEach(btn => {
                if (btn) {
                    btn.addEventListener('click', openModal);
                }
            });

            closeBtn.addEventListener('click', closeModal);
            
            modal.addEventListener('click', (e) => {
                if (e.target === modal) {
                    closeModal();
                }
            });

            document.addEventListener('keydown', (e) => {
                if (e.key === 'Escape' && !modal.classList.contains('hidden')) {
                    closeModal();
                }
            });

            // Enhanced form validation
            const emailForm = document.getElementById('email-signin-form');
            const emailInput = document.getElementById('email');
            const passwordInput = document.getElementById('password');
            const emailError = document.getElementById('email-error');
            const passwordError = document.getElementById('password-error');

            function validateEmail(email) {
                const vitEmailRegex = /^[a-zA-Z0-9._%+-]+@vit\.ac\.in$/;
                return vitEmailRegex.test(email);
            }

            function validatePassword(password) {
                return password.length >= 6;
            }

            emailInput.addEventListener('blur', () => {
                if (!validateEmail(emailInput.value)) {
                    emailError.classList.remove('hidden');
                    emailInput.classList.add('border-red-500');
                } else {
                    emailError.classList.add('hidden');
                    emailInput.classList.remove('border-red-500');
                }
            });

            passwordInput.addEventListener('blur', () => {
                if (!validatePassword(passwordInput.value)) {
                    passwordError.classList.remove('hidden');
                    passwordInput.classList.add('border-red-500');
                } else {
                    passwordError.classList.add('hidden');
                    passwordInput.classList.remove('border-red-500');
                }
            });

            emailForm.addEventListener('submit', (e) => {
                e.preventDefault();
                
                const isEmailValid = validateEmail(emailInput.value);
                const isPasswordValid = validatePassword(passwordInput.value);
                
                if (isEmailValid && isPasswordValid) {
                    alert('Sign-in successful! Welcome to BookSwap VIT!');
                    closeModal();
                }
            });

            document.getElementById('google-signin').addEventListener('click', () => {
                alert('Google Sign-in would redirect to OAuth flow');
            });

            document.getElementById('phone-signin').addEventListener('click', () => {
                alert('Phone sign-in would show OTP verification form');
            });
        }
